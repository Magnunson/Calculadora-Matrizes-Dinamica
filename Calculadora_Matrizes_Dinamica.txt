var
   // Começamos por difinir uma Matrizes globais com limite máximo 10x10
   mA, mB, mRes: vetor [1..10, 1..10] de real

   //Criamos as  Variáveis de controle e dimensões das matrizes globais
   L, C, L2, C2, i, j, k, opcao: inteiro
   det, det_inv: real

inicio
    //Usamos uma estrutura de repetição para garantir que o usuario digite a
    //opção valida
   repita
   //apresentamos o menu de escolha
      escreval("")
      escreval("======= CALCULADORA DE MATRIZES (MÁX 10x10) =======")
      escreval("1 - Somar Matrizes")
      escreval("2 - Subtrair Matrizes")
      escreval("3 - Determinante (Apenas 3x3)")
      escreval("4 - Matriz Transposta")
      escreval("5 - Multiplicar Matrizes (A x B)")
      escreval("6 - Matriz Inversa (Apenas 2x2)")
      escreval("0 - Sair")
      escreval("")
      escreva("Escolha uma opção: ")
      leia(opcao)
        escreval("")
      escolha opcao
      //Aqui somamos, se o usuario escolher a opção 1 e subtraimos de escolher 2
      caso 1, 2
      //pedimos ao usuario o numro de linha e columa da matriz
         escreval("--- Soma/Subtração de Matrizes ---")
         escreva("Digite a quantidade de linhas: ")
         leia(L)
          escreval("")
         escreva("Digite a quantidade de colunas: ")
         leia(C)
         //preenchimento da matriz A
         escreval("Preencha Matriz A:")
         para i de 1 ate L faca
            para j de 1 ate C faca
               escreva("A [", i, ",", j, "]: ")
               leia(mA[i,j])
            fimpara
         fimpara
          //preenchimento da matriz B
         escreval("Preencha Matriz B:")
         para i de 1 ate L faca
            para j de 1 ate C faca
               escreva("B [", i, ",", j, "]: ")
               leia(mB[i,j])
            fimpara
         fimpara

         escreval("--- RESULTADO ---")
         para i de 1 ate L faca
            para j de 1 ate C faca
               se opcao = 1 entao
                  mRes[i,j] <- mA[i,j] + mB[i,j]
               senao
                  mRes[i,j] <- mA[i,j] - mB[i,j]
               fimse
               escreva(mRes[i,j]:6:2, " ")
            fimpara
            escreval("")
         fimpara
         //Calculamos a determinante da matriz 3x3 usando o metodo de Lapluss
      caso 3
         escreval("--- Determinante 3x3 ---")
         // ela elimina a primeira linha e trabalha com a segunda linha
         para i de 1 ate 3 faca
            para j de 1 ate 3 faca
               escreva("Matriz [", i, ",", j, "]: ")
               leia(mA[i,j])
            fimpara
         fimpara
         det <- (mA[1,1] * ((mA[2,2] * mA[3,3]) - (mA[2,3] * mA[3,2]))) - (mA[1,2] * ((mA[2,1] * mA[3,3]) - (mA[2,3] * mA[3,1]))) + (mA[1,3] * ((mA[2,1] * mA[3,2]) - (mA[2,2] * mA[3,1])))
         escreval("O Determinante é: ", det:6:2)
       //Sabemos que a matriz transposta é calculada da seguinte forma:
       //se a matriz A é difinida por 2x3, a sua matrix transposta sera 3x2
      caso 4
         escreval("--- Matriz Transposta ---")
         escreva("Digite o numero de Linhas: ")
         leia(L)
          escreval("")
         escreva("Digite o numero de Colunas: ")
         leia(C)

         para i de 1 ate L faca
            para j de 1 ate C faca
               escreva("Elemento [", i, ",", j, "]: ")
               leia(mA[i,j])
            fimpara
         fimpara

         escreval("--- MATRIZ TRANSPOSTA RESULTANTE ---")
         para i de 1 ate C faca
            para j de 1 ate L faca
               escreva(mA[j,i]:6:2, " ")
            fimpara
            escreval("")
         fimpara
      //Na multiplição, não se pode multiplicar duas matrizes se
      //o numero de linha da matriz A, não for igual ao numero de coluna da
      //matriz B, entao usamos estruturas de condições para verificar
      caso 5
         escreval("--- Multiplicação de Matrizes ---")
         escreva("Linhas de A: ")
         leia(L)
         escreva("Colunas de A: ")
         leia(C)
          escreval("")
         escreva("Linhas de B: ")
         leia(L2)
         escreva("Colunas de B: ")
         leia(C2)
         //condição para a multiplição
         se C <> L2 entao
            escreval("Erro: Colunas de A devem ser iguais às Linhas de B!")
         senao
            escreval("Preencha a matriz A:")
            para i de 1 ate L faca
               para j de 1 ate C faca
                  escreva("A[", i, ",", j, "]: ")
                  leia(mA[i,j])
               fimpara
            fimpara
            escreval("Preencha a matriz B:")
            para i de 1 ate L2 faca
               para j de 1 ate C2 faca
                  escreva("B[", i, ",", j, "]: ")
                  leia(mB[i,j])
               fimpara
            fimpara

            escreval("---O RESULTADO DAS MATRIZES A x B É: ---")
            para i de 1 ate L faca
               para j de 1 ate C2 faca
                  mRes[i,j] <- 0
                  para k de 1 ate C faca
//Aqui primeiro multiplicamos os elementos da matriz A com os elementos da
//matriz B, e posteriormente fazemos a soma do resultado do produto das
//matrizes A e B,
                     mRes[i,j] <- mRes[i,j] + (mA[i,k] * mB[k,j])
                  fimpara
                  escreva(mRes[i,j]:6:2, " ")
               fimpara
               escreval("")
            fimpara
         fimse

      caso 6
         escreval("--- Matriz Inversa (2x2) ---")
         escreval("Preencha a Matriz 2x2:")
         para i de 1 ate 2 faca
            para j de 1 ate 2 faca
               escreva("[", i, ",", j, "]: ")
               leia(mA[i,j])
            fimpara
         fimpara

         det_inv <- (mA[1,1] * mA[2,2]) - (mA[1,2] * mA[2,1])

         se det_inv = 0 entao
            escreval("Esta matriz não possui inversa (Determinante = 0)")
         senao
            escreval("--- MATRIZ INVERSA ---")
            // Aqi usamos a regra da algebra para achar a inversa de uma matriz
            //: 1/det * [d -b; -c a]
            escreva((mA[2,2]/det_inv):6:2, " ")
            escreva((-mA[1,2]/det_inv):6:2)
            escreval("")
            escreva((-mA[2,1]/det_inv):6:2, " ")
            escreva((mA[1,1]/det_inv):6:2)
            escreval("")
         fimse

      caso 0
         escreval("Saindo...")
      outrocaso
         escreval("Opção Inválida! Por favor digite uma opção valida de 0 a 6 ")
      fimescolha

   ate opcao = 0
